Задача 1
from math import cos
N = int(input())
chast_sum = 0
for i in range(1, N +1):
    funct = (2 + cos(i))/(i**2+4)
    chast_sum += funct
print(chast_sum)

Задача 2
A = int(input())
B = int(input())
for i in range(A, B + 1):
    copy_n = i
    result = 0
    while (i!=0) :
        ost = i % 10
        result = result * 10 + ost
        i = i//10
    if result == copy_n:
        print(result)
